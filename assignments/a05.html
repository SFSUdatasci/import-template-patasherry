Introduction

This report analyzes socioecological monitoring data from restoration sites in the Sacramento-San Joaquin Delta to explore visitor patterns by location and visitor type.

Data

The data come from the EDI Data Repository: Sacramento-San Joaquin Delta Socioecological Monitoring, downloaded on 4/1/2026 and saved to the data/ folder.

Analysis

Preparing the data

# Pivot visitor type columns from wide to long format for plotting
visits_long <- delta_visits %>%
    pivot_longer(cols = c(sm_boat, med_boat, lrg_boat, bank_angler, scientist, cars),
                 names_to = "visitor_type",
                 values_to = "quantity") %>%
    # Rename location column to something shorter
    rename(restore_loc = eco_restore_approximate_location) %>%
    # Drop the notes column
    select(-notes)

# Calculate total daily visits grouped by location, date, and visitor type
daily_visits_loc <- visits_long %>%
    group_by(restore_loc, date, visitor_type) %>%
    summarise(daily_visits = sum(quantity, na.rm = TRUE)) %>%
    ungroup()

Total visits by location and visitor type

# Define a reusable custom theme based on theme_bw
my_theme <- theme_bw(base_size = 16) +
    # Move legend to bottom and remove y-axis tick marks
    theme(legend.position = "bottom",
          axis.ticks.y = element_blank())

# Add a column for total visits per location, used for reordering
daily_visits_totals <- daily_visits_loc %>%
    group_by(restore_loc) %>%
    mutate(n = sum(daily_visits)) %>%
    ungroup() %>%
    # Reorder restore_loc factor levels by total visits (ascending)
    mutate(restore_loc = fct_reorder(restore_loc, n))

# Build a stacked horizontal bar chart of visits by location and visitor type
ggplot(data = daily_visits_totals,
       # Map location to y-axis, daily visits to x-axis, color bars by visitor type
       aes(x = daily_visits, y = restore_loc,
           fill = visitor_type)) +
    # Draw stacked columns
    geom_col() +
    # Use colorblind-friendly viridis color palette for fill
    scale_fill_viridis_d() +
    # Set x-axis breaks every 20 units and remove gap at axis origin
    scale_x_continuous(breaks = seq(0, 120, 20), expand = c(0, 0)) +
    # Add axis labels, legend label, title, and subtitle
    labs(x = "Number of Visits",
         y = "Restoration Location",
         fill = "Type of Visitor",
         title = "Total Number of Visits to Delta Restoration Areas by Visitor Type",
         subtitle = "Sum of all visits during study period") +
    # Apply the custom theme
    my_theme

Faceted plots by restoration location

# Build a faceted bar chart — one panel per restoration location
facet_plot <- ggplot(data = daily_visits_totals,
       # Map visitor type to x and y, color bars by visitor type
       aes(x = visitor_type, y = daily_visits,
           fill = visitor_type)) +
    # Draw columns for each visitor type
    geom_col() +
    # Create a separate panel for each restoration location, free y-axis scales
    facet_wrap(~restore_loc,
               scales = "free_y",
               ncol = 5,
               nrow = 2) +
    # Use colorblind-friendly viridis palette
    scale_fill_viridis_d() +
    # Add axis and title labels
    labs(x = "Type of Visitor",
         y = "Number of Visits",
         title = "Total Number of Visits to Delta Restoration Areas",
         subtitle = "Sum of all visits during study period") +
    # Use black-and-white theme
    theme_bw() +
    # Move legend to bottom, remove x-axis ticks and text for cleanliness
    theme(legend.position = "bottom",
          axis.ticks.x = element_blank(),
          axis.text.x = element_blank())

# Print the plot
facet_plot

Interactive table of sampling locations

# Get one row per location with its coordinates
locations <- visits_long %>%
    distinct(restore_loc, .keep_all = TRUE) %>%
    select(restore_loc, latitude, longitude)

# Display as an interactive searchable table
datatable(locations)

Interactive map of restoration sites

# Initialize a leaflet map using the locations data frame
leaflet(locations) %>%
    # Add a topographic base map layer
    addProviderTiles(providers$Esri.WorldTopoMap) %>%
    # Add circle markers at each site's latitude and longitude
    addCircleMarkers(
        lng = ~longitude,        # map longitude column to x position
        lat = ~latitude,         # map latitude column to y position
        popup = ~restore_loc,    # show location name on click
        radius = 5,              # set circle size
        fillColor = "salmon",    # set fill color
        fillOpacity = 1,         # make fill fully opaque
        stroke = TRUE,           # draw a border around each circle
        weight = 0.5,            # set border thickness
        color = "white",         # set border color
        opacity = 1              # make border fully opaque
    )

Discussion

Prospect and Grizzly Bay had the most visits, with medium boats being the most common visitor type. Most other sites had lower visitation. The map shows all 10 sites clustered within the Delta region.
